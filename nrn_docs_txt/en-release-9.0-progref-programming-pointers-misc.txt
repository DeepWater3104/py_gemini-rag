Return a pointer (coded as a double) to the segment
of the currently accessed section that
contains location
x
. If you wish to compute a segment number
index where 1 is the first nonzero area segment and nseg is the last

func segnum() {
        if ($1 <= 0) {
                return 0
        }else if ($1 >= 1) {
                return nseg+1
        }else {
                return int($1*nseg + .5)
        }
}
